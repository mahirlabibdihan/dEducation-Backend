CREATE TABLE Users (
  user_id NUMBER GENERATED ALWAYS as IDENTITY(START with 1 INCREMENT by 1),
  name VARCHAR2(100) NOT NULL,
  image VARCHAR2(1000),
  email VARCHAR2(100) NOT NULL,
  pass VARCHAR2(1024) NOT NULL,
  type VARCHAR2(1024) NOT NULL
);

CREATE TABLE Students (
	student_id NUMBER REFERENCES Users(user_id),
	gender VARCHAR2(100),
	date_of_birth DATE,
	instituition VARCHAR2(100),
	phone_number NUMBER,
	address_id NUMBER REFERENCES Address(address_id),
);

CREATE TABLE Address (
	address_id NUMBER,
	city VARCHAR2(100),
	district VARCHAR2(100)
);
