  id NUMBER GENERATED ALWAYS as IDENTITY(START with 1 INCREMENT by 1),
  name VARCHAR2(100) NOT NULL,
  image VARCHAR2(1000),
  email VARCHAR2(100) NOT NULL,
  pass VARCHAR2(1024) NOT NULL
);
